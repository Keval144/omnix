# NLP Pipeline

## Overview

Natural Language Processing (NLP) deals with text data. This guide covers the complete pipeline from raw text to model predictions, including text classification, sentiment analysis, and language modeling.

## Pipeline Steps

```
1. Data Loading
2. Text Cleaning
3. Tokenization
4. Text Preprocessing
5. Vectorization/Embedding
6. Feature Engineering
7. Model Training
8. Evaluation
```

## Text Cleaning

### Basic Text Cleaning

```python
import re
import string

def clean_text(text):
    """Basic text cleaning pipeline."""
    if pd.isna(text):
        return ""
    
    text = str(text)
    
    # Lowercase
    text = text.lower()
    
    # Remove URLs
    text = re.sub(r'http\S+|www\S+|https\S+', '', text)
    
    # Remove HTML tags
    text = re.sub(r'<.*?>', '', text)
    
    # Remove special characters and numbers (optional)
    text = re.sub(r'[^a-zA-Z\s]', '', text)
    
    # Remove extra whitespace
    text = re.sub(r'\s+', ' ', text).strip()
    
    return text

# Apply cleaning
df['text_clean'] = df['text'].apply(clean_text)

print("Sample cleaned text:")
print(df['text_clean'].iloc[0])
```

### Advanced Text Cleaning

```python
import unicodedata

def advanced_clean_text(text):
    """Advanced text cleaning with normalization."""
    if pd.isna(text):
        return ""
    
    text = str(text)
    
    # Normalize unicode characters
    text = unicodedata.normalize('NFKD', text).encode('ascii', 'ignore').decode('ascii')
    
    # Lowercase
    text = text.lower()
    
    # Remove URLs
    text = re.sub(r'http\S+|www\S+|https\S+', '', text)
    
    # Remove email addresses
    text = re.sub(r'\S+@\S+', '', text)
    
    # Remove mentions and hashtags
    text = re.sub(r'@\w+|#\w+', '', text)
    
    # Remove HTML entities
    text = re.sub(r'&\w+;', '', text)
    
    # Remove numbers (keep if important)
    text = re.sub(r'\d+', '', text)
    
    # Remove punctuation
    text = text.translate(str.maketrans('', '', string.punctuation))
    
    # Remove extra whitespace
    text = ' '.join(text.split())
    
    return text

df['text_clean'] = df['text'].apply(advanced_clean_text)
```

## Stopword Removal

```python
import nltk
nltk.download('stopwords')

from nltk.corpus import stopwords

stop_words = set(stopwords.words('english'))

def remove_stopwords(text):
    words = text.split()
    words = [w for w in words if w not in stop_words]
    return ' '.join(words)

df['text_no_stop'] = df['text_clean'].apply(remove_stopwords)

# Custom stopwords
custom_stopwords = stop_words | {'would', 'could', 'also', 'one'}
df['text_custom_stop'] = df['text_clean'].apply(
    lambda x: ' '.join([w for w in x.split() if w not in custom_stopwords])
)
```

## Stemming and Lemmatization

```python
import nltk
nltk.download('wordnet')
nltk.download('omw-1.4')

from nltk.stem import PorterStemmer, LancasterStemmer, WordNetLemmatizer

stemmer = PorterStemmer()
lemmatizer = WordNetLemmatizer()

text = "running runners ran easily"

# Stemming (faster, less accurate)
stemmed = [stemmer.stem(word) for word in text.split()]
print(f"Stemmed: {stemmed}")
# ['run', 'runner', 'ran', 'easili']

# Lemmatization (slower, more accurate)
lemmatized = [lemmatizer.lemmatize(word) for word in text.split()]
print(f"Lemmatized: {lemmatized}")
# ['running', 'runner', 'ran', 'easily']

# With POS tagging (more accurate)
nltk.download('averaged_perceptron_tagger')
nltk.download('averaged_perceptron_tagger_eng')

def lemmatize_with_pos(text):
    doc = nlp(text)
    return ' '.join([token.lemma_ for token in doc])

df['text_lemmatized'] = df['text_no_stop'].apply(lemmatize_with_pos)
```

### TF-IDF (Term Frequency-Inverse Document Frequency)

```python
from sklearn.feature_extraction.text import TfidfVectorizer

# TF-IDF Vectorizer
tfidf = TfidfVectorizer(
    max_features=5000,
    min_df=2,           # Ignore terms appearing in less than 2 documents
    max_df=0.95,        # Ignore terms appearing in more than 95% of documents
    ngram_range=(1, 2), # Unigrams and bigrams
    stop_words='english'
)

X = tfidf.fit_transform(df['text_clean'])

print(f"Vocabulary size: {len(tfidf.vocabulary_)}")
print(f"Feature matrix shape: {X.shape}")

# TF-IDF scores for a document
sample_tfidf = X[0].toarray()[0]
top_indices = sample_tfidf.argsort()[-5:][::-1]
print(f"Top TF-IDF terms: {[feature_names[i] for i in top_indices]}")
```

# Average word vectors for document
def document_vector(doc, model):
    doc = [word for word in doc.split() if word in model.wv]
    if len(doc) == 0:
        return np.zeros(100)

## NLP Classification Pipeline

### Traditional ML Approach (TF-IDF + Classifier)

```python
from sklearn.model_selection import train_test_split
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.svm import LinearSVC
from sklearn.metrics import classification_report, accuracy_score

# Split data
X_train, X_test, y_train, y_test = train_test_split(
    df['text_clean'], df['label'], test_size=0.2, random_state=42, stratify=df['label']
)

# TF-IDF Vectorization
tfidf = TfidfVectorizer(max_features=5000, ngram_range=(1, 2))
X_train_tfidf = tfidf.fit_transform(X_train)
X_test_tfidf = tfidf.transform(X_test)

# Train Logistic Regression
lr = LogisticRegression(max_iter=1000, random_state=42)
lr.fit(X_train_tfidf, y_train)
y_pred_lr = lr.predict(X_test_tfidf)

print("Logistic Regression Results:")
print(classification_report(y_test, y_pred_lr))

# Train SVM
svm = LinearSVC(random_state=42)
svm.fit(X_train_tfidf, y_train)
y_pred_svm = svm.predict(X_test_tfidf)

print("SVM Results:")
print(classification_report(y_test, y_pred_svm))
```

### Perplexity (Language Models)

```python
# Calculate perplexity for language models
import math

def calculate_perplexity(model, tokenizer, text):
    encodings = tokenizer(text, return_tensors="pt")
    input_ids = encodings['input_ids']
    
    with torch.no_grad():
        outputs = model(input_ids, labels=input_ids)
        loss = outputs.loss
    
    perplexity = math.exp(loss.item())
    return perplexity

perplexity = calculate_perplexity(model, tokenizer, "This is a test sentence.")
print(f"Perplexity: {perplexity:.2f}")
```

## Complete NLP Pipeline Example

```python
import pandas as pd
import numpy as np
import re
from sklearn.model_selection import train_test_split
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression
from sklearn.pipeline import Pipeline
from sklearn.metrics import classification_report, accuracy_score
import nltk
from nltk.corpus import stopwords

# Download NLTK data
nltk.download('punkt')
nltk.download('stopwords')

# 1. Load data
df = pd.read_csv('text_classification.csv')

# 2. Text preprocessing function
def preprocess_text(text):
    if pd.isna(text):
        return ""
    
    text = str(text).lower()
    text = re.sub(r'http\S+', '', text)  # Remove URLs
    text = re.sub(r'@\w+', '', text)     # Remove mentions
    text = re.sub(r'#\w+', '', text)    # Remove hashtags
    text = re.sub(r'[^a-zA-Z\s]', '', text)  # Keep only letters
    text = re.sub(r'\s+', ' ', text).strip()  # Remove extra spaces
    
    # Tokenize and remove stopwords
    stop_words = set(stopwords.words('english'))
    words = text.split()
    words = [w for w in words if w not in stop_words and len(w) > 2]
    
    return ' '.join(words)

# 3. Preprocess
df['text_clean'] = df['text'].apply(preprocess_text)

# 4. Split data
X_train, X_test, y_train, y_test = train_test_split(
    df['text_clean'], df['label'],
    test_size=0.2, random_state=42, stratify=df['label']
)

# 5. Create pipeline
pipeline = Pipeline([
    ('tfidf', TfidfVectorizer(
        max_features=5000,
        ngram_range=(1, 2),
        min_df=2
    )),
    ('classifier', LogisticRegression(max_iter=1000, random_state=42))
])

# 6. Train
pipeline.fit(X_train, y_train)

# 7. Evaluate
y_pred = pipeline.predict(X_test)

print(f"Accuracy: {accuracy_score(y_test, y_pred):.4f}")
print("\nClassification Report:")
print(classification_report(y_test, y_pred))

# 8. Predict new text
new_text = ["This is an amazing product! Highly recommend."]
prediction = pipeline.predict(new_text)
print(f"\nPrediction: {prediction}")
```

## Common Pitfalls

### Text Cleaning Mistakes

```python
# BAD: Removing all punctuation
# Loses important semantic information (emoticons, exclamation marks)
text = re.sub(r'[^\w\s]', '', text)

# GOOD: Selective cleaning
# Keep emoticons if relevant for sentiment analysis
text = re.sub(r'http\S+', '', text)  # Only remove URLs
```

### Data Leakage

```python
# BAD: TF-IDF on entire dataset before split
tfidf = TfidfVectorizer()
X_tfidf = tfidf.fit_transform(df['text'])  # LEAKAGE!
X_train, X_test = train_test_split(X_tfidf)

# GOOD: Fit only on training data
X_train, X_test, y_train, y_test = train_test_split(df['text'], df['label'])
tfidf = TfidfVectorizer()
X_train_tfidf = tfidf.fit_transform(X_train)
X_test_tfidf = tfidf.transform(X_test)  # Transform, don't fit!
```

### Ignoring Class Imbalance

```python
# BAD: Not handling imbalanced classes
model = LogisticRegression()
model.fit(X_train, y_train)

# GOOD: Use class weights or oversampling
model = LogisticRegression(class_weight='balanced')
model.fit(X_train, y_train)
```

## Best Practices

- [ ] Clean text appropriately for your task
- [ ] Handle missing/empty text
- [ ] Use stratified splits for classification
- [ ] Consider TF-IDF vs embeddings based on task
- [ ] Use pretrained models for small datasets
- [ ] Monitor for data leakage
- [ ] Use appropriate evaluation metrics
- [ ] Consider domain-specific preprocessing
- [ ] Test with real-world text variations
- [ ] Monitor for model drift over time
