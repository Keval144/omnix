# Time Series Pipeline

## Overview

Time series data has temporal ordering where time is the primary dimension. This guide covers forecasting, trend analysis, seasonality detection, and building predictive models for sequential data.

## Pipeline Steps

```
1. Data Loading and Time Parsing
2. Exploratory Time Series Analysis
3. Handling Missing Timestamps
4. Time-Based Feature Engineering
5. Stationarity Analysis
6. Train/Test Split (Time-Based)
7. Model Training
8. Evaluation and Forecasting
```

### Detect Missing Timestamps

```python
# Check for gaps in time series
date_range = pd.date_range(start=df.index.min(), end=df.index.max(), freq='D')
missing_dates = date_range.difference(df.index)

```

### Fill Missing Timestamps

```python
# Create complete date range and reindex
date_range = pd.date_range(start=df.index.min(), end=df.index.max(), freq='D')
df = df.reindex(date_range)

# Method 1: Forward fill (use previous value)
df['value_ffill'] = df['value'].fillna(method='ffill')

# Method 2: Backward fill (use next value)
df['value_bfill'] = df['value'].fillna(method='bfill')

# Method 3: Linear interpolation
df['value_interp'] = df['value'].interpolate(method='linear')

# Method 4: Time-based interpolation
df['value_time_interp'] = df['value'].interpolate(method='time')

# Method 5: Rolling mean imputation
df['value_rolling'] = df['value'].fillna(df['value'].rolling(7, min_periods=1).mean())

# Choose best method based on data characteristics
print(df[['value', 'value_ffill', 'value_interp']].head(20))
```

### Handle Missing with Forward Fill + Known Values

```python
# For production systems with regular updates
def fill_missing_ts(data, max_fill=7):
    """Forward fill with limit, then backward fill remaining."""
    data = data.copy()
    data = data.fillna(method='ffill', limit=max_fill)
    data = data.fillna(method='bfill')  # Fill any remaining
    return data

df['value_clean'] = fill_missing_ts(df['value'])
```

### Cyclical Encoding

```python
# Encode cyclical patterns with sin/cos
def encode_cyclical(df, col, period):
    df[f'{col}_sin'] = np.sin(2 * np.pi * df[col] / period)
    df[f'{col}_cos'] = np.cos(2 * np.pi * df[col] / period)
    return df

df = encode_cyclical(df, 'month', 12)
```

## Stationarity Analysis

### Augmented Dickey-Fuller Test

```python
from statsmodels.tsa.stattools import adfuller

def adf_test(series, name=''):
    """Augmented Dickey-Fuller test for stationarity."""
    result = adfuller(series.dropna(), autolag='AIC')
    
    print(f'ADF Test for {name}:')
    print(f'  ADF Statistic: {result[0]:.4f}')
    print(f'  p-value: {result[1]:.4f}')
    print(f'  Critical Values:')
    for key, value in result[4].items():
        print(f'    {key}: {value:.4f}')
    
    if result[1] <= 0.05:
        print('  Result: Stationary (reject null hypothesis)')
    else:
        print('  Result: Non-stationary (fail to reject null hypothesis)')
    
    return result[1]

p_value = adf_test(df['value'], 'value')
```

### KPSS Test

```python
from statsmodels.tsa.stattools import kpss

def kpss_test(series, name=''):
    """KPSS test for stationarity."""
    result = kpss(series.dropna(), regression='c', nlags='auto')
    
    print(f'KPSS Test for {name}:')
    print(f'  KPSS Statistic: {result[0]:.4f}')
    print(f'  p-value: {result[1]:.4f}')
    print(f'  Critical Values:')
    for key, value in result[3].items():
        print(f'    {key}: {value:.4f}')
    
    if result[1] <= 0.05:
        print('  Result: Non-stationary (reject null hypothesis)')
    else:
        print('  Result: Stationary (fail to reject null hypothesis)')

kpss_test(df['value'], 'value')
```

### LSTM Neural Network

```python
# pip install tensorflow

import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout
from sklearn.preprocessing import MinMaxScaler

# Scale data
scaler = MinMaxScaler()
scaled_train = scaler.fit_transform(train['value'].values.reshape(-1, 1))

# Create sequences
def create_sequences(data, seq_length):
    X, y = [], []
    for i in range(len(data) - seq_length):
        X.append(data[i:(i + seq_length)])
        y.append(data[i + seq_length])
    return np.array(X), np.array(y)

seq_length = 12
X_train_seq, y_train_seq = create_sequences(scaled_train, seq_length)
X_test_seq, y_test_seq = create_sequences(scaled_test, seq_length)

# Build LSTM model

    LSTM(50, activation='relu', input_shape=(seq_length, 1)),
    Dropout(0.2),
    Dense(1)
])

model.compile(optimizer='adam', loss='mse')

# Train
model.fit(X_train_seq, y_train_seq, epochs=50, batch_size=32, validation_split=0.1, verbose=0)

# Forecast
```

## Evaluation Metrics

```python
from sklearn.metrics import mean_squared_error, mean_absolute_error
import numpy as np

def evaluate_forecast(y_true, y_pred, model_name='Model'):
    """Evaluate forecast performance."""
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    mae = mean_absolute_error(y_true, y_pred)
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100
    
    print(f"=== {model_name} ===")
    print(f"  RMSE: {rmse:.4f}")
    print(f"  MAE: {mae:.4f}")
    print(f"  MAPE: {mape:.2f}%")
    
    return {'rmse': rmse, 'mae': mae, 'mape': mape}

# Compare all models
results = {}
for name, forecast in [('Naive', forecast_naive), ('SES', forecast_ses), 
                        ('Holt', forecast_holt), ('HW', forecast_hw),
                        ('ARIMA', forecast_arima), ('XGBoost', forecast_xgb)]:
    results[name] = evaluate_forecast(test['value'], forecast, name)
```

## Complete Pipeline Example

```python
import pandas as pd
import numpy as np
from statsmodels.tsa.holtwinters import ExponentialSmoothing
from statsmodels.tsa.arima.model import ARIMA
from sklearn.metrics import mean_squared_error
import matplotlib.pyplot as plt

# 1. Load and parse dates
df = pd.read_csv('time_series.csv', parse_dates=['date'], index_col='date')
df = df.sort_index()

# 2. Handle missing values
df = df.asfreq('D')  # Ensure daily frequency
df['value'] = df['value'].interpolate(method='time')

# 3. Feature engineering
df['month'] = df.index.month
df['dayofweek'] = df.index.dayofweek
df['lag_1'] = df['value'].shift(1)
df['rolling_mean_7'] = df['value'].rolling(7).mean()

# 4. Train/test split
train_size = int(len(df) * 0.8)
train, test = df.iloc[:train_size], df.iloc[train_size:]

# 5. Train multiple models
models = {}

# Holt-Winters
hw_model = ExponentialSmoothing(
    train['value'].dropna(),
    trend='add',
    seasonal='add',
    seasonal_periods=7
).fit()
models['Holt-Winters'] = hw_model.forecast(len(test))

# ARIMA
arima_model = ARIMA(train['value'].dropna(), order=(5, 1, 2)).fit()
models['ARIMA'] = arima_model.forecast(len(test))

# 6. Evaluate and select best
results = {}
for name, forecast in models.items():
    rmse = np.sqrt(mean_squared_error(test['value'], forecast))
    results[name] = rmse
    print(f"{name} RMSE: {rmse:.4f}")

best_model = min(results, key=results.get)
print(f"\nBest model: {best_model}")

# 7. Forecast future
future_periods = 30
final_model = models[best_model.replace(' ', '-').lower()]
future_forecast = final_model.forecast(future_periods)
```

## Common Pitfalls

### Not Using Time-Based Split

```python
# BAD: Random split breaks temporal order
from sklearn.model_selection import train_test_split
X_train, X_test = train_test_split(df, test_size=0.2)  # WRONG!

# GOOD: Time-based split
train = df.iloc[:train_size]
test = df.iloc[train_size:]
```

### Data Leakage

```python
# BAD: Using future information in features
df['future_value'] = df['value'].shift(-1)  # LEAKAGE!

# GOOD: Only use past information
df['lag_1'] = df['value'].shift(1)  # OK
```

### Ignoring Stationarity

```python
# BAD: Using non-stationary data directly
# Results in spurious correlations

# GOOD: Make series stationary first
df['value_diff'] = df['value'].diff()  # First difference
```

### Wrong Evaluation

```python
# BAD: Using R² for time series
from sklearn.metrics import r2_score
r2_score(y_test, y_pred)  # Can be negative for time series!

# GOOD: Use RMSE, MAE, MAPE
rmse = np.sqrt(mean_squared_error(y_test, y_pred))
```

## Best Practices

- [ ] Always sort by time before analysis
- [ ] Infer and set proper frequency
- [ ] Handle missing timestamps explicitly
- [ ] Visualize trends, seasonality, and residuals
- [ ] Test for stationarity (ADF, KPSS)
- [ ] Use time-based split (no shuffling)
- [ ] Engineer lag and rolling features carefully
- [ ] Consider multiple forecasting methods
- [ ] Evaluate with appropriate metrics
- [ ] Monitor for concept drift in production
