# Data Cleaning

## Overview

Data cleaning is the first and most critical step in any ML pipeline. Poor quality data leads to poor quality models regardless of algorithm choice.

## Missing Values

### Detecting Missing Values

```python
import pandas as pd
import numpy as np

# Check missing values
missing_counts = df.isnull().sum()
missing_pct = (df.isnull().sum() / len(df)) * 100

# Create summary dataframe
missing_summary = pd.DataFrame({
    'missing_count': missing_counts,
    'missing_pct': missing_pct
}).sort_values('missing_pct', ascending=False)

print(missing_summary[missing_summary['missing_pct'] > 0])
```

### Imputation Strategies

#### Mean Imputation (Numeric Features)

**When to use:** Data is normally distributed without significant outliers.

```python
# BAD: Dropping rows with missing values (loses data)
df_clean = df.dropna(subset=['feature'])

# GOOD: Mean imputation
df['feature'] = df['feature'].fillna(df['feature'].mean())
```

**Complete Example:**
```python
# Mean imputation for numeric columns
numeric_cols = df.select_dtypes(include=[np.number]).columns

for col in numeric_cols:
    if df[col].isnull().sum() > 0:
        df[col] = df[col].fillna(df[col].mean())
```

#### Median Imputation (Numeric Features)

**When to use:** Data has outliers or is skewed.

```python
# Median is robust to outliers
df['feature'] = df['feature'].fillna(df['feature'].median())
```

**Example with Skewed Data:**
```python
import numpy as np

# Income data is often right-skewed (few high earners)
# Mean would pull up, median is more representative

print(f"Mean: {df['income'].mean()}")      # $85,000 (skewed by outliers)
print(f"Median: {df['income'].median()}")  # $52,000 (more representative)

df['income'] = df['income'].fillna(df['income'].median())
```

#### Mode Imputation (Categorical Features)

```python
# Most frequent value imputation
df['category'] = df['category'].fillna(df['category'].mode()[0])
```

#### Forward Fill / Backward Fill (Time Series)

```python
# Use previous value (forward fill)
df['feature'] = df['feature'].fillna(method='ffill')

# Use next value (backward fill)
df['feature'] = df['feature'].fillna(method='bfill')
```

### Advanced Imputation Methods

#### KNN Imputation

```python
from sklearn.impute import KNNImputer

# More accurate than simple imputation
imputer = KNNImputer(n_neighbors=5)

# Select numeric columns for imputation
numeric_df = df.select_dtypes(include=[np.number])
df_imputed = pd.DataFrame(
    imputer.fit_transform(numeric_df),
    columns=numeric_df.columns,
    index=numeric_df.index
)

df[numeric_df.columns] = df_imputed
```

#### Iterative Imputation (Multivariate)

```python
from sklearn.experimental import enable_iterative_imputer
from sklearn.impute import IterativeImputer

imputer = IterativeImputer(max_iter=10, random_state=42)
df_imputed = pd.DataFrame(
    imputer.fit_transform(df.select_dtypes(include=[np.number])),
    columns=df.select_dtypes(include=[np.number]).columns
)
```

### Missing Value Decision Tree

```
Is the feature important for the model?
├── YES: Try to recover the value
│   ├── Time series? → Forward/backward fill
│   ├── Related to other features? → KNN or Iterative imputation
│   └── No clear pattern? → Mean/Median/Mode
└── NO: Consider dropping the feature
    └── If >50-70% missing → Drop feature
```

### Fuzzy Matching

```python
from difflib import get_close_matches

def standardize_category(value, valid_categories):
    """Standardize to closest valid category."""
    if pd.isna(value):
        return value
    
    value_lower = str(value).strip().lower()
    
    # Direct match
    if value_lower in valid_categories:
        return value_lower
    
    # Fuzzy match
    matches = get_close_matches(value_lower, valid_categories, n=1, cutoff=0.6)
    if matches:
        return matches[0]
    
    return value_lower  # Keep original if no match

valid_cats = ['small', 'medium', 'large']
df['size'] = df['size'].apply(lambda x: standardize_category(x, valid_cats))
```

## Outlier Detection

### IQR Method (Interquartile Range)

```python
def detect_outliers_iqr(df, column, multiplier=1.5):
    """
    Detect outliers using IQR method.
    
    IQR = Q3 - Q1
    Lower bound = Q1 - (multiplier * IQR)
    Upper bound = Q3 + (multiplier * IQR)
    
    Default multiplier 1.5: Standard outliers
    Multiplier 3.0: Extreme outliers
    """
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1
    
    lower_bound = Q1 - (multiplier * IQR)
    upper_bound = Q3 + (multiplier * IQR)
    
    outliers = df[(df[column] < lower_bound) | (df[column] > upper_bound)]
    
    return outliers, lower_bound, upper_bound

# Example usage
outliers, lower, upper = detect_outliers_iqr(df, 'price')
print(f"Lower bound: {lower}, Upper bound: {upper}")
print(f"Outliers found: {len(outliers)}")
```

### Z-Score Method

```python
from scipy import stats

def detect_outliers_zscore(df, column, threshold=3):
    """
    Detect outliers using Z-score method.
    
    Z-score = (value - mean) / std
    Values with |Z-score| > threshold are outliers
    """
    z_scores = np.abs(stats.zscore(df[column], nan_policy='omit'))
    outliers = df[z_scores > threshold]
    
    return outliers, z_scores

# Example usage
outliers, z_scores = detect_outliers_zscore(df, 'price', threshold=3)
print(f"Outliers found: {len(outliers)}")
print(f"Max Z-score: {z_scores.max():.2f}")
```

## Common Mistakes

### ❌ Don't Do These

1. **Don't drop all rows with any missing value**
   ```python
   # WRONG: Loses too much data
   df_clean = df.dropna()
   
   # RIGHT: Impute or drop strategically
   df_clean = df.dropna(thresh=len(df.columns)*0.7)  # Keep rows with 70%+ data
   ```

2. **Don't fill missing values with 0 (unless appropriate)**
   ```python
   # WRONG: 0 can bias your model
   df['age'] = df['age'].fillna(0)
   
   # RIGHT: Use mean/median/mode or NaN for algorithms that handle it
   df['age'] = df['age'].fillna(df['age'].median())
   ```

3. **Don't remove all outliers**
   ```python
   # WRONG: Outliers might be important signal
   df_clean = df[(np.abs(stats.zscore(df['value'])) < 1)]
   
   # RIGHT: Investigate outliers first, then decide
   ```

4. **Don't assume missing = not applicable**
   ```python
   # WRONG: Different missing reasons need different handling
   df['income'] = df['income'].fillna(0)
   
   # RIGHT: Understand WHY values are missing
   ```

## Best Practices Checklist

- [ ] Identify all missing values and their percentages
- [ ] Understand WHY data is missing (MCAR, MAR, MNAR)
- [ ] Choose imputation strategy based on data distribution
- [ ] Remove true duplicates (check for legitimate duplicates first)
- [ ] Convert all columns to appropriate data types
- [ ] Standardize categorical values (lowercase, strip whitespace)
- [ ] Detect and handle outliers (investigate before removing)
- [ ] Document all cleaning decisions
- [ ] Keep raw data separate from cleaned data
- [ ] Create a reproducible cleaning pipeline

### Date Parsing Edge Cases
```python
# Handle various date formats
def parse_dates(series):
    formats = [
        '%Y-%m-%d',
        '%m/%d/%Y',
        '%d-%m-%Y',
        '%Y/%m/%d',
        '%B %d, %Y',
        None  # Let pandas infer
    ]
    
    for fmt in formats:
        try:
            return pd.to_datetime(series, format=fmt, errors='raise')
        except:
            continue
    
    return pd.to_datetime(series, errors='coerce')

df['date'] = parse_dates(df['date'])
```

## Pipeline Integration

```python
from sklearn.base import BaseEstimator, TransformerMixin
from sklearn.pipeline import Pipeline

class DataCleaner(BaseEstimator, TransformerMixin):
    def __init__(self, numeric_strategy='median', categorical_strategy='mode'):
        self.numeric_strategy = numeric_strategy
        self.categorical_strategy = categorical_strategy
        
    def fit(self, X, y=None):
        self.numeric_cols = X.select_dtypes(include=['number']).columns
        self.categorical_cols = X.select_dtypes(include=['object', 'category']).columns
        
        self.numeric_medians = X[self.numeric_cols].median()
        self.categorical_modes = X[self.categorical_cols].mode().iloc[0]
        
        return self
    
    def transform(self, X):
        X = X.copy()
        
        # Convert to numeric (coerce errors)
        for col in X.select_dtypes(include=['object']).columns:
            try:
                X[col] = pd.to_numeric(X[col])
            except:
                pass
        
        # Re-identify numeric columns after conversion
        numeric_cols = X.select_dtypes(include=['number']).columns
        
        # Impute numeric
        for col in numeric_cols:
            X[col] = X[col].fillna(self.numeric_medians.get(col, X[col].median()))
        
        # Impute categorical
        for col in X.select_dtypes(include=['object', 'category']).columns:
            X[col] = X[col].fillna(self.categorical_modes.get(col, X[col].mode()[0]))
        
        return X

# Usage in pipeline
pipeline = Pipeline([
    ('cleaner', DataCleaner()),
    ('scaler', StandardScaler()),
    ('model', RandomForestRegressor())
])
```
